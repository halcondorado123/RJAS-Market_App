CREATE DATABASE RJAS_DB
GO

USE RJAS_DB
GO

CREATE SCHEMA RJAS
GO

/* Creaci√≥n de tablas */
CREATE TABLE RJAS.CATEGORIA_PRODUCTO(
	ID_CATEGORIA INT IDENTITY(1,1) NOT NULL,
	NOMBRE_CATEGORIA VARCHAR(100) NOT NULL
)
GO

CREATE TABLE RJAS.PROVEEDORES(
	ID_PROVEEDOR INT IDENTITY(1,1) NOT NULL,
	NOMBRE_PROVEEDOR VARCHAR(100) NOT NULL,
	ID_CATEGORIA INT NOT NULL,
	TELEFONO VARCHAR(14) NOT NULL, 
	EMAIL VARCHAR(100) NOT NULL,
	DESCRIPCION_PRODUCTO NVARCHAR(250) NULL,
	CANTIDAD INT NOT NULL,
	PRECIO DECIMAL(10, 2) NOT NULL,
	IMPUESTO_IVA DECIMAL(10, 2) NOT NULL,
	DESCUENTO INT NULL,
	ES_ACTIVO BIT NULL,
	TOTAL DECIMAL(10, 2) NOT NULL
)
GO

-- Crear las llaves primarias
ALTER TABLE RJAS.CATEGORIA_PRODUCTO
ADD CONSTRAINT PK_CATEGORIA_PRODUCTO PRIMARY KEY (ID_CATEGORIA)
GO

ALTER TABLE RJAS.PROVEEDORES
ADD CONSTRAINT PK_PROVEEDORES PRIMARY KEY (ID_PROVEEDOR)
GO

-- Crear llave foranea
ALTER TABLE RJAS.PROVEEDORES
ADD CONSTRAINT FK_PROVEEDORES_CATEGORIA
FOREIGN KEY (ID_CATEGORIA) REFERENCES RJAS.CATEGORIA_PRODUCTO(ID_CATEGORIA);
GO
